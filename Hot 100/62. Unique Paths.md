```c++
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> dp(n, vector<int>(m,0));

        if (n <= 1 || m <= 1) return 1;
        
        for (int i = 0; i < m; i++) {
            dp[0][i] = 1;
        }

        for (int i = 0; i < n; i++) {
            dp[i][0] = 1;
        }

        //start from second row
        for (int i = 1; i < n; i++){
            for (int j = 1; j < m; j++){
                dp[i][j]= dp[i-1][j] + dp[i][j-1];
            }
        }
        return dp[n-1][m-1]; 
    }
};
```

超出时间限制，BFS
```c++
class Solution {
public:
    int paths = 0;

    int uniquePaths(int m, int n) {
        move(m, n, 1, 1);
        return paths;
    };

    void move (int m, int n, int cur_m, int cur_n){
        if (cur_m == m && cur_n == n)  {
            paths += 1;
        }
        else if (cur_m <  m && cur_n <  n){
            move(m, n, cur_m +1, cur_n);
            move(m, n, cur_m, cur_n +1);
        }
        else if (cur_m < m && cur_n == n){
            move(m, n, cur_m+1, cur_n);
        }
        else{
            move(m, n, cur_m, cur_n +1);
        }
        return;
    };
};
```
