class Solution {
public:
    void sortColors(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return;
        int l = 0;
        int r = n-1;
        int i = 0;
        while (i <= r){
            while (l < n &&  nums[l] == 0)l++;
            while (r >= 0 && nums[r] == 2)r--;
            if (l == n || r == -1 || l > r ) return;
            i = l > i?l:i;

            if (nums[i] == 0){
                nums[i] = nums[l];
                nums[l] = 0;
                l++;
            }
            else if (nums[i] == 2){
                nums[i] = nums[r];
                nums[r] = 2;
                r--;
            }
            else{
                i++;
            }
        }
    }
};
