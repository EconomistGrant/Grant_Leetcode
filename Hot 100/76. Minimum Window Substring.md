class Solution {
public:
    string minWindow(string s, string t) {
        int n = s.length();
        if (n == 0) return "";
        string window = s;

        int char_included = 0;
        unordered_set<char> lookup;
        unordered_map<char, int> count;

        for (int i = 0; i < t.length();i++){
            lookup.insert(t[i]);
            count[t[i]] = 0;
        }
        int m = lookup.size();

        //check s
        for (int i = 0 ; i < n ; i++){
            if(lookup.find(s[i]) != lookup.end()){
                if (count[s[i]] == 0) char_included++;
                count[s[i]]++;
            }
        }
        if (char_included < m) return"";

        char_included = 0;
        count.clear();
        if (lookup.find(s[0]) != lookup.end()) {
            count[s[0]] = 1;
            char_included++;
        }

        int l = 0;
        int r = 0;
        while (l <= n){
            if (char_included < m && r != n - 1){
                r++;
                if (lookup.find(s[r]) != lookup.end()){
                    if (count[s[r]] == 0) char_included++;
                    count[s[r]] = count[s[r]]+1;
                }
            }
            else{
                if (char_included == m && r - l + 1 < window.size()) {
                    window = s.substr(l, r - l + 1);
                }
                if(lookup.find(s[l]) != lookup.end()){
                    count[s[l]]--;
                    if (count[s[l]] == 0) char_included--;
                }
                l++;
            }
        }

        return window;
    }
};
